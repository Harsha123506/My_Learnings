Async validators – real-world use case
Async validators are for anything that needs I/O or time (server, network, IndexedDB, etc.)

export function usernameTakenValidator(
  api: UserApi
): AsyncValidatorFn {
  return (control: AbstractControl) => {
    if (!control.value) {
      return of(null);
    }

    return api.isUsernameTaken(control.value).pipe(
      map(isTaken => (isTaken ? { usernameTaken: true } : null)),
      catchError(() => of(null)) // don’t block form on API error
    );
  };
}
